function loadStream(url, drmKeyId = '', drmKey = '') { const enableDRM = document.getElementById('enable-drm').checked; let config = { file: url, width: "100%", height: "100%", aspectratio: "16:9" }; if (enableDRM) { // Ensure DRM config is properly set config.drm = { clearkey: { keyId: drmKeyId, key: drmKey } }; } jwplayer("player-container").setup(config); }
